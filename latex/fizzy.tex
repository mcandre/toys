\usepackage{ifthen}
\usepackage{forloop}
\usepackage{fmtcount}
\usepackage{intcalc}
\usepackage{multicol}

% Margins
\setlength{\topmargin}{-0.25in}
\pagestyle{empty}

\begin{document}
\begin{multicols}{2}
\newcounter{i}
\forloop{i}{1}{\value{i} < 101}{
	\noindent
	\ifthenelse{\equal{\intcalcMod{\value{i}}{15}}{0}}{
		FizzBuzz
	}{
		\ifthenelse{\equal{\intcalcMod{\value{i}}{3}}{0}}{
			Fizz
		}{
			\ifthenelse{\equal{\intcalcMod{\value{i}}{5}}{0}}{
				Buzz
			}{
				\decimal{i}
			}
		}
	}\\
}
\end{multicols}
\end{document}
