\noindent
$
A = \{ a, set(a) \}
\\\\
B = P(A)
\\\\
|A| = x
\\\\
|B| = 2^x
\\\\
x > 1
\\\\
2^x > x
\\\\
|B| > |A|
\\\\
A \neq \{ a, set(a) \} 
$
